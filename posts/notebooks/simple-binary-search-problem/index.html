def special_min(lst):
    if len(lst) == 1:
        return lst[0]
    mid = (len(lst)-1) // 2
    if lst[mid] > lst[-1]:
        return special_min(lst[mid+1:])
    elif lst[0] >= lst[mid]:
        return special_min(lst[:mid+1])
    else:
        return lst[0]
