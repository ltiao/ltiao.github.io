class Rectangle:
    def __init__(self, p1, p2):
        self.p1 = p1
        self.p2 = p2

    @classmethod
    def from_dimensions(cls, p1, width, height=None):
        if height is None:
            height = width
        return cls(p1, Point(p1.x+width, p1.y+height))
