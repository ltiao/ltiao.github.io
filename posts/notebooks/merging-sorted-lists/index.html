def _merge_it(xs, ys):
    
    if not xs:
        return ys
    if not ys:
        return xs
    
    i = j = 0
    ret = []
    
    while i < len(xs) and j < len(ys):
        if xs[i] < ys[j]:
            ret.append(xs[i])
            i += 1
        else:
            ret.append(ys[j])
            j += 1
            
    ret.extend(xs[i:])
    ret.extend(ys[j:])
    
    return ret

def _merge_all(*lsts):
    
    h = map(lambda x: (x[1][0], x[0]), enumerate(lsts))
    indices = [1] * len(lsts)
    heapify(h)
    while any(map(lambda i, lst: i < len(lst), indices, lsts)):
        v, i = heappop(h)
        heappush(h, (lsts[i][indices[i]], i))
        indices[i] += 1
        yield v
    while h:
        v, _ = heappop(h)
        yield v

def _merge_all(*iterables):
    
    h = []
    for it in map(iter, iterables):
        v = next(it, None)
        if v is not None:
            h.append((v, it))
    heapify(h)

    while h:
        v, it = heappop(h)
        yield v
        w = next(it, None)
        if w is not None:
            heappush(h, (w, it))
