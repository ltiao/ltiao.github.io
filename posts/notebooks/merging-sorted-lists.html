def _merge_re(xs, ys):
    
    if not xs:
        return ys
    if not ys:
        return xs
    
    x, y = xs[0], ys[0]
    
    if x < y:
        ret = [x]
        xs  = xs[1:]
    else:
        ret = [y]
        ys  = ys[1:]
        
    ret.extend(merge(xs, ys))
    
    return ret

def _merge_it(xs, ys):
    
    if not xs:
        return ys
    if not ys:
        return xs
    
    i = j = 0
    ret = []
    
    while i < len(xs) and j < len(ys):
        if xs[i] < ys[j]:
            ret.append(xs[i])
            i += 1
        else:
            ret.append(ys[j])
            j += 1
            
    ret.extend(xs[i:])
    ret.extend(ys[j:])
    
    return ret
